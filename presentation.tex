\section{Contexte et approche}
% =============================================================

\begin{frame}{Pourquoi le Machine Learning Quantique en finance ?}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \textbf{L'hypothèse quantique}\\[6pt]
      Les circuits quantiques encodent les données
      dans un espace de très haute dimension,
      potentiellement \textbf{inaccessible aux algorithmes classiques}.

      \bigskip
      En pratique : capturer des corrélations non-linéaires
      dans les données de risque (crédit, marché, fraude).

      \bigskip
      \begin{block}{Notre question centrale}
        Est-ce que ça apporte un avantage \emph{mesurable}
        sur des données financières réelles aujourd'hui ?
      \end{block}

    \column{0.48\textwidth}
      \textbf{Notre méthode : QMKL}\\[6pt]
      \textbf{Quantum Multiple Kernel Learning} combine
      plusieurs \emph{kernels quantiques} — chacun encode
      les données différemment dans un circuit quantique.

      \bigskip
      \begin{enumerate}
        \item Les données client sont encodées en qubits
        \item On calcule la similarité quantique entre clients
        \item Un SVM classe les clients selon cette similarité
        \item On combine 12 kernels différents pour plus de robustesse
      \end{enumerate}
  \end{columns}
\end{frame}

\begin{frame}{Les 3 datasets financiers testés}
  \begin{columns}[T]
    \column{0.32\textwidth}
      \begin{block}{German Credit}
        \textbf{Risque de crédit}\\[3pt]
        \small
        1\,000 clients · 20 variables\\
        30\,\% de défauts\\[4pt]
        Tâche : prédire si un\\
        client va faire défaut\\[4pt]
        \textit{Cas d'usage bancaire direct}
      \end{block}

    \column{0.32\textwidth}
      \begin{block}{Bank Marketing}
        \textbf{Score marketing}\\[3pt]
        \small
        45\,000 contacts · 16 variables\\
        11\,\% de succès (déséquilibré)\\[4pt]
        Tâche : prédire si un\\
        client souscrit un dépôt\\[4pt]
        \textit{Ciblage client, fidélisation}
      \end{block}

    \column{0.32\textwidth}
      \begin{block}{Breast Cancer}
        \textbf{Benchmark médical}\\[3pt]
        \small
        569 patients · 30 variables\\
        Bien séparable (AUC $>$ 0.99)\\[4pt]
        Utilisé comme référence :\\
        mesurer la limite haute\\[4pt]
        \textit{Cas favorable au quantique}
      \end{block}
  \end{columns}

  \bigskip
  \begin{center}
    \small
    Métrique : \textbf{AUC} (aire sous la courbe ROC)
    \quad \textbf{0.5} = aléatoire \quad \textbf{1.0} = parfait\\
    Tous les résultats : moyenne sur \textbf{20 répétitions}
    avec des splits aléatoires différents
  \end{center}
\end{frame}

\begin{frame}{Comment les données sont encodées dans un circuit quantique}
  \begin{columns}[T]
    \column{0.54\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F_circuit.png}
    \column{0.44\textwidth}
      \textbf{PauliFeatureMap ZZ} — le cœur de l'encodage\\[8pt]
      \begin{enumerate}\small
        \item \textbf{Superposition} (portes H) : chaque qubit
              explore simultanément 0 et 1
        \item \textbf{Encodage individuel} $R_z(2\alpha x_i)$ :
              chaque variable financière tourne un qubit
        \item \textbf{Encodage des interactions} $R_z(2\alpha x_i x_j)$ :
              corrélations \emph{non-linéaires} via l'intrication
      \end{enumerate}
      \medskip
      \begin{block}{Similarité quantique}
        \small
        $K(x,x') = |\langle\psi(x')|\psi(x)\rangle|^2$\\[2pt]
        Clients proches $\Rightarrow$ états quantiques similaires
      \end{block}
      \medskip
      \small $\alpha$ : \textbf{largeur de bande} du kernel
      (analogue au $\gamma$ du RBF classique).
  \end{columns}
\end{frame}

\begin{frame}{Les 12 kernels quantiques : 6 familles d'encodage}
  \begin{center}
    \includegraphics[width=0.96\linewidth]{results/presentation/pres_F_kernels.png}
  \end{center}
  \vspace{-4pt}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \small
      Chaque \textbf{famille} correspond à un choix d'opérateurs de Pauli
      qui détermine \emph{quelles} corrélations sont capturées.\\[4pt]
      Pour chaque famille : \textbf{2 valeurs de $\alpha$}
      (petit = bande large, grand = bande étroite).
    \column{0.48\textwidth}
      \small
      \begin{block}{Pourquoi 12 kernels ?}
        Aucun kernel unique n'est optimal sur tous les datasets.
        QMKL les \textbf{combine} pour obtenir une représentation
        plus riche qu'un seul circuit.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Comment fonctionne chaque stratégie QMKL ?}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{block}{Average — Moyenne uniforme}
        \footnotesize $K_{mkl} = \tfrac{1}{M}\sum_m K_m$ \hfill \good{Le plus simple}\\[1pt]
        Tous les kernels ont le même poids.\\
        Aucun paramètre à ajuster.
      \end{block}
      \smallskip
      \begin{block}{Centered Alignment}
        \footnotesize Poids $w_m^*$ calculés analytiquement\\
        en maximisant $\langle K_{mkl},\, yy^\top \rangle_F$.\\
        \good{Solution fermée, sans optimisation.}
      \end{block}
      \smallskip
      \begin{block}{Bayesian Optimization}
        \footnotesize Poids $w_m \in [0,1]$ optimisés par cross-val.\\
        \bad{200$\times$ plus lent} pour +0.8 pt vs Average.
      \end{block}

    \column{0.48\textwidth}
      \begin{block}{QMKL-Boost \textnormal{\footnotesize(notre contribution)}}
        \footnotesize Sélection \textbf{binaire} par QUBO :\\
        $\min_{w\in\{0,1\}^M}\!-\!\sum_m w_m s_m + \lambda\!\sum_{m\neq m'}\! w_m w_{m'} A_{mm'}$\\[2pt]
        $s_m$ = AUC kernel $m$ \quad $A_{mm'}$ = redondance\\
        \good{+4.1 pts AUC, 2 kernels sur 12 sélectionnés}
      \end{block}
      \smallskip
      \begin{block}{VQKL — Variationnel}
        \footnotesize Poids appris par gradient descent.\\
        \bad{Surapprentissage} sur N=200 — pire qu'Average.
      \end{block}
      \smallskip
      \begin{alertblock}{Enseignement clé}
        \footnotesize La complexité algorithmique ne paie pas.\\
        L'avantage est dans la \textbf{diversité des kernels},\\
        pas dans l'optimisation des poids.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Quels kernels apportent le plus ? (Analyse Shapley)}
  \begin{columns}[T]
    \column{0.57\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F8_shapley.png}

    \column{0.41\textwidth}
      \textbf{Lecture}\\[6pt]
      La valeur de Shapley mesure la \emph{contribution}
      marginale de chaque kernel à la performance QMKL.
      Un kernel avec contribution négative \textbf{dégrade}
      la combinaison.

      \bigskip
      \textbf{Pourquoi ZZ $\alpha$=4.0 domine ?}\\[4pt]
      C'est le kernel \textbf{le plus différent} des autres —
      il apporte une information complémentaire
      que les autres kernels ne couvrent pas.

      \bigskip
      \begin{block}{Règle clé}
        \small $r = 0.738$ entre diversité des kernels
        et gain QMKL.\\
        \textbf{Diversité $>$ quantité.}
      \end{block}
  \end{columns}
\end{frame}

\section{Notre contribution : QMKL-Boost}
% =============================================================

\begin{frame}{QMKL-Boost — sélection intelligente des kernels}
  \begin{columns}[T]
    \column{0.44\textwidth}
      \textbf{Le problème}\\[4pt]
      \small Moyenner 12 kernels uniformément dilue
      l'information utile avec du bruit des kernels
      redondants ou contre-productifs.

      \medskip
      \textbf{Notre approche}\\[4pt]
      \small Un \textbf{algorithme QUBO} — même famille
      que les optimiseurs QPU Pasqal / IBM —
      sélectionne les \textbf{2 kernels les plus
      complémentaires} parmi les 12.

      \medskip
      \begin{exampleblock}{Résultats (N=150)}
        \small
        German Credit : \textbf{+4.1 pts AUC}\\
        Bank Marketing : \textbf{+7.6 pts AUC}\\[2pt]
        2 kernels sélectionnés sur 12 :\\
        \texttt{ZZ a=2.5} + \texttt{XZ a=0.5}
      \end{exampleblock}

    \column{0.54\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F2_qmkl_boost.png}
  \end{columns}
\end{frame}

\begin{frame}{Comment QUBO trouve les bons kernels}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F10_lambda_sweep.png}

    \column{0.43\textwidth}
      \textbf{Le paramètre $\lambda$ contrôle}\\
      \textbf{le compromis diversité / performance}\\[6pt]
      \begin{itemize}
        \item $\lambda=0$ : sélectionne tous les kernels\\
              avec le meilleur score individuel\\
              (parfois redondants)
        \item $\lambda=1$ : \good{optimal} — 2 kernels de\\
              familles différentes, complémentaires
        \item $\lambda$ trop grand : 1 seul kernel,\\
              trop peu d'information
      \end{itemize}

      \bigskip
      \textbf{Kernels sélectionnés à $\lambda^*=1$ :}\\[4pt]
      \small \texttt{ZZ a=2.5} (entrelacement fort) +\\
      \texttt{XZ a=0.5} (encodage mixte)\\[4pt]
      $\Rightarrow$ deux familles différentes,\\
      \phantom{$\Rightarrow$} diversité maximale avec seulement 2 kernels.
  \end{columns}
\end{frame}

\section{Limites : concentration et matériel}
% =============================================================

\begin{frame}{La limite fondamentale : barren plateaus}
  \begin{columns}[T]
    \column{0.57\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F9_concentration.png}

    \column{0.41\textwidth}
      \textbf{Qu'est-ce qu'un barren plateau ?}\\[6pt]
      À mesure qu'on ajoute des qubits, les valeurs
      du kernel quantique se concentrent vers zéro —
      les données deviennent \textbf{indiscernables}
      pour le circuit.

      \medskip
      \begin{itemize}
        \item De Q=2 à Q=8 :\\
              \bad{$-60\,\%$} de variance utile
        \item Les kernels locaux réduisent\\
              ce problème de 40\,\%
        \item Problème \textbf{ouvert} en recherche
      \end{itemize}

      \medskip
      \begin{alertblock}{Pour la banque}
        \small Augmenter le nombre de qubits
        ne garantit \emph{pas} d'améliorer la performance.
      \end{alertblock}
  \end{columns}
\end{frame}

      \small Backend : \textbf{IBM Torino} (Heron r2, 133 qubits)\\

      \medskip
      \begin{alertblock}{\small Verdict}
        \small Résultats \textbf{incohérents} entre kernels.\\
        Comportement imprévisible sur QPU réel.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Ce que ça veut dire pour la banque}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \textbf{Aujourd'hui — Ne pas déployer}\\[6pt]
      \begin{itemize}
        \item \bad{$-7$ pts AUC} sur le risque crédit :\\
              les classiques sont supérieurs et moins coûteux
        \item Hardware trop bruité :\\
              variance $\pm 0.20$ inacceptable en production
        \item Plus de données n'efface pas l'écart :\\
              limite structurelle, pas statistique
      \end{itemize}

      \bigskip
      \textbf{Recommandation immédiate}\\[4pt]
      Continuer à utiliser \textbf{RBF-SVM} et
      \textbf{Forêt aléatoire} en production.
      Ne pas investir dans du hardware QPU
      pour la classification de crédit.

    \column{0.48\textwidth}
      \textbf{Horizon 3--5 ans — À surveiller}\\[6pt]
      \begin{itemize}
        \item \good{QMKL-Boost} : notre approche QUBO
              gagne \good{+4 pts} sur German Credit
              — à suivre quand le hardware sera stable
        \item \good{Interprétabilité réglementaire} :\\
              QUBO sélectionne 2 kernels sur 12,
              modèle plus explicable (Bâle IV, RGPD)
        \item \good{Vitesse d'inférence} : 2 kernels =
              6$\times$ moins de calcul à la décision
      \end{itemize}

      \bigskip
      \begin{exampleblock}{Position recommandée}
        \small R\&D exploratoire, budget contenu.
        Réévaluation complète en 2028--2030
        quand la correction d'erreur QPU sera mature.
      \end{exampleblock}
  \end{columns}
\end{frame}
