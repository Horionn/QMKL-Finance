\section{Contexte et approche}
% =============================================================

\begin{frame}{Pourquoi le Machine Learning Quantique en finance ?}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \textbf{L'hypothèse quantique}\\[6pt]
      Les circuits quantiques encodent les données
      dans un espace de très haute dimension,
      potentiellement \textbf{inaccessible aux algorithmes classiques}.

      \bigskip
      En pratique : capturer des corrélations non-linéaires
      dans les données de risque (crédit, marché, fraude).

      \bigskip
      \begin{block}{Notre question centrale}
        Est-ce que ça apporte un avantage \emph{mesurable}
        sur des données financières réelles aujourd'hui ?
      \end{block}

    \column{0.48\textwidth}
      \textbf{Notre méthode : QMKL}\\[6pt]
      \textbf{Quantum Multiple Kernel Learning} combine
      plusieurs \emph{kernels quantiques} — chacun encode
      les données différemment dans un circuit quantique.

      \bigskip
      \begin{enumerate}
        \item Les données client sont encodées en qubits
        \item On calcule la similarité quantique entre clients
        \item Un SVM classe les clients selon cette similarité
        \item On combine 12 kernels différents pour plus de robustesse
      \end{enumerate}
  \end{columns}
\end{frame}

\begin{frame}{Du tableau bancaire au circuit quantique : le pipeline}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \textbf{4 étapes de transformation}\\[5pt]

      \begin{block}{\small \textbf{1.} Encodage catégoriel}
        \footnotesize
        Variables textuelles (type de crédit, emploi\ldots)\\
        $\rightarrow$ \textbf{one-hot encoding} (\texttt{get\_dummies}, \texttt{drop\_first})\\
        German Credit : 20 vars $\rightarrow$ 47 colonnes numériques
      \end{block}
      

      \begin{block}{\small \textbf{2.} Réduction de dimension (PCA)}
        \footnotesize
        47 colonnes $\rightarrow$ \textbf{Q composantes principales}\\
        \textbf{Q = nombre de qubits} : chaque feature occupe un qubit\\
        Variance expliquée conservée : $\approx$85\,\% avec Q=6
      \end{block}
      

      \begin{block}{\small \textbf{3.} Normalisation en deux temps}
        \footnotesize
        \textbf{MinMaxScaler} : mise à l'échelle vers $[0,\,2]$\\
        Résultat : $x_i \in [0, 2]$ pour tout $i$
      \end{block}

    \column{0.48\textwidth}
      \textbf{Pourquoi ces choix ?}\\[6pt]
      \begin{itemize}\small
        \item \textbf{PCA obligatoire} : un circuit à Q qubits
              ne peut encoder que Q valeurs simultanément.
              La PCA sélectionne les directions les plus informatives.

        \bigskip
        \item \textbf{Plage $[0,\,2]$} : les portes quantiques
              sont des rotations d'angle $\theta = 2*a x_i$.
              Avec $a \in \{0.5,\,4.0\}$ et $x_i \in [0,\,2]$,
              on explore $[0,\,4\pi]$ — une révolution complète.

        \bigskip
        \item \textbf{Standardisation d'abord} : évite que
              les variables à grande variance (revenus, montants)
              saturent l'espace de rotation et écrasent
              les signaux faibles.
      \end{itemize}

      \medskip
      \begin{exampleblock}{Exemple German Credit, Q=4}
        \footnotesize
        20 vars $\xrightarrow{\text{one-hot}}$ 47
        $\xrightarrow{\text{PCA}}$ 4
        $\xrightarrow{\text{scaler}}$ $x_i \in [0,2]$\\
        Train : 268 clients $\times$ 4 features
      \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Comment les données sont encodées dans un circuit quantique}
  \begin{columns}[T]
    \column{0.54\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F_circuit.png}
    \column{0.44\textwidth}
      \textbf{PauliFeatureMap ZZ} — le cœur de l'encodage\\[8pt]
      \begin{enumerate}\small
        \item \textbf{Superposition} (portes H) : chaque qubit
              explore simultanément 0 et 1
        \item \textbf{Encodage individuel} $R_z(2 a x_i)$ :
              chaque variable financière tourne un qubit
        \item \textbf{Encodage des interactions} $R_z(2 a x_i x_j)$ :
              corrélations \emph{non-linéaires} via l'intrication
      \end{enumerate}
      \medskip
      \begin{block}{Similarité quantique}
        \small
        $K(x,x') = |\langle\psi(x')|\psi(x)\rangle|^2$\\[2pt]
        Clients proches $\Rightarrow$ états quantiques similaires
      \end{block}
      
  \end{columns}
\end{frame}

\begin{frame}{Les 12 kernels quantiques : géométrie de similarité}
  \begin{center}
    \includegraphics[width=0.97\linewidth]{results/presentation/pres_F_kernels.png}
  \end{center}
  \vspace{-4pt}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \small
      \textbf{Lire une heatmap kernel :} cases bleues = clients similaires
      (même région de l'espace quantique) ; cases blanches = clients dissemblables.
      Le bloc en haut à gauche (x\textsubscript{1}–x\textsubscript{3}) révèle
      la structure locale capturée par chaque famille.\\[3pt]
      \textbf{Hiérarchie d'interaction :}
      \begin{itemize}\small
        \item \textcolor{blue}{Local (1-corps)} — Z : effets individuels
        \item \textcolor{red}{Pairwise (2-corps)} — ZZ, XZ : corrélations paires
        \item \textcolor{violet}{3-corps non-comm.} — YXX, YZX : interactions croisées
      \end{itemize}
    \column{0.46\textwidth}
      \small
      \begin{block}{Pourquoi 12 kernels ?}
        Aucun kernel unique n'est optimal sur tous les datasets.
        QMKL les \textbf{combine} pour obtenir une représentation
        plus riche qu'un seul circuit.
      \end{block}
      \vspace{3pt}
      \begin{exampleblock}{Lien Shapley}
        \small Les familles YXX et ZZ contribuent le plus
        (valeur Shapley $> +0.03$) : elles apportent
        une information \textbf{complémentaire} absente des autres.
      \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Comment fonctionne chaque stratégie QMKL ?}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \begin{block}{Average — Moyenne uniforme}
        \footnotesize $K_{mkl} = \tfrac{1}{M}\sum_m K_m$ \hfill \good{Le plus simple}\\[1pt]
        Tous les kernels ont le même poids.\\
        Aucun paramètre à ajuster.
      \end{block}
      \smallskip
      \begin{block}{Centered Alignment}
        \footnotesize Poids $w_m^*$ calculés analytiquement\\
        en maximisant $\langle K_{mkl},\, yy^\top \rangle_F$.\\
        \good{Solution fermée, sans optimisation.}
      \end{block}
      \smallskip
      \begin{block}{Bayesian Optimization}
        \footnotesize Poids $w_m \in [0,1]$ optimisés par cross-val.\\
        
      \end{block}

    \column{0.48\textwidth}
      \begin{block}{QMKL-Boost \textnormal{\footnotesize(notre contribution)}}
        \footnotesize Sélection \textbf{binaire} par QUBO :\\
        $\min_{w\in\{0,1\}^M}\!-\!\sum_m w_m s_m + \lambda\!\sum_{m\neq m'}\! w_m w_{m'} A_{mm'}$\\[2pt]
        $s_m$ = AUC kernel $m$ \quad $A_{mm'}$ = redondance\\
      \end{block}
      \smallskip
      \begin{block}{VQKL — Variationnel}
        \footnotesize Poids appris par gradient descent.\\
      \end{block}
      
      
  \end{columns}
\end{frame}

\section{Notre contribution : QMKL-Boost}
% =============================================================

\begin{frame}{QMKL-Boost — sélection intelligente des kernels}
  \begin{columns}[T]
    \column{0.44\textwidth}
      \textbf{Le problème}\\[4pt]
      \small Moyenner 12 kernels uniformément dilue
      l'information utile avec du bruit des kernels
      redondants ou contre-productifs.

      \medskip
      \textbf{Notre approche}\\[4pt]
      \small Un \textbf{algorithme QUBO} — même famille
      que les optimiseurs QPU Pasqal / IBM —
      sélectionne les \textbf{2 kernels les plus
      complémentaires} parmi les 12.

      \medskip
      \begin{exampleblock}{Résultats (N=150)}
        \small
        German Credit : \textbf{+4.1 pts AUC}\\
        Bank Marketing : \textbf{+7.6 pts AUC}\\[2pt]
        2 kernels sélectionnés sur 12 :\\
        \texttt{ZZ a=2.5} + \texttt{XZ a=0.5}
      \end{exampleblock}

    \column{0.54\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F2_qmkl_boost.png}
  \end{columns}
\end{frame}

\begin{frame}{La matrice Q — comment QUBO choisit les kernels}
  \begin{columns}[T]
    \column{0.58\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_qubo_matrix.png}

    \column{0.40\textwidth}
      \textbf{Structure de la matrice Q}\\[5pt]
      \begin{itemize}
        \item \textbf{Diagonale $Q_{mm} = -s_m$} :\\
              valeur négative (bleue) = encourager\\
              la sélection du kernel $m$\\
              (plus $s_m$ est élevé, plus il est attractif)

        \medskip
        \item \textbf{Hors-diagonale $Q_{mm'} = \lambda A_{mm'}$} :\\
              valeur positive (rouge) = pénaliser\\
              la co-sélection de kernels redondants\\
              ($A_{mm'}$ = corrélation de Frobenius)
      \end{itemize}

      \medskip
      \begin{exampleblock}{Résultat à $\lambda^*=1$}
        \small\texttt{ZZ $\alpha$=4.0} (entrelacement fort)\\
        $+$ \texttt{XZ $\alpha$=0.5} (encodage mixte)\\[2pt]
        Deux familles différentes, $A_{mm'} \approx 0.20$\\
        AUC $= \mathbf{0.823}$
      \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Quand le quantique devient-il indispensable pour QUBO ?}
  \begin{columns}[T]
    \column{0.56\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_qubo_complexity.png}

    \column{0.42\textwidth}
      \textbf{L'explosion combinatoire $2^M$}\\[5pt]
      \begin{itemize}
        \item $M = 12$ kernels :\\
              $2^{12} = $ 4\,096 sous-ensembles\\
              \good{classique parfaitement viable}

        \item $M = 25$ kernels :\\
              $2^{25} \approx $ 33 millions — \amb{limite classique}

        \item $M = 50$ kernels :\\
              $2^{50} \approx 10^{15}$ — \bad{classique impossible}
      \end{itemize}

      \medskip
      \begin{exampleblock}{Ce n'est pas la simulation quantique}
        \small C'est l'\textbf{optimisation combinatoire} qui devient
        indispensable. Les annealers quantiques (D-Wave, Pasqal)
        résolvent QUBO en $O(M^{1.5})$ empirique —
        break-even vers $M \approx 25$.
      \end{exampleblock}
  \end{columns}
\end{frame}

\section{Limites : concentration et matériel}
% =============================================================

\begin{frame}{La limite fondamentale : barren plateaus}
  \begin{columns}[T]
    \column{0.57\textwidth}
      \includegraphics[width=\linewidth]{results/presentation/pres_F9_concentration.png}

    \column{0.41\textwidth}
      \textbf{Qu'est-ce qu'un barren plateau ?}\\[6pt]
      À mesure qu'on ajoute des qubits, les valeurs
      du kernel quantique se concentrent vers zéro —
      les données deviennent \textbf{indiscernables}
      pour le circuit.

      \medskip
      \begin{itemize}
        \item De Q=2 à Q=8 :\\
              \bad{$-60\,\%$} de variance utile
        \item Les kernels locaux réduisent\\
              ce problème de 40\,\%
        \item Problème \textbf{ouvert} en recherche
      \end{itemize}

  \end{columns}
\end{frame}

              limite structurelle, pas statistique
      \end{itemize}

      \

    \column{0.48\textwidth}
      \textbf{À surveiller}\\[6pt]
      \begin{itemize}
        \item \good{QMKL-Boost} : notre approche QUBO
              gagne \good{+4 pts} sur German Credit
              — à suivre quand le hardware sera stable
        \item \good{Interprétabilité réglementaire} :\\
              QUBO sélectionne 2 kernels sur 12,
              modèle plus explicable 
        \item \good{Vitesse d'inférence} : 2 kernels =
              6$\times$ moins de calcul à la décision
      \end{itemize}
